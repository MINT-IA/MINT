```dart
/// screen_pump — dual-device golden test helper for MINT.
///
/// Wraps a widget in a MaterialApp + MediaQuery + Localizations
/// matching either iPhone 14 Pro or Galaxy A14 dimensions.
///
/// Usage:
///   testWidgets('s4 mtc default', (tester) async {
///     await pumpScreen(
///       tester,
///       device: GoldenDevice.iphone14Pro,
///       child: const MyScreen(),
///     );
///     await expectLater(
///       find.byType(MyScreen),
///       matchesGoldenFile('goldens/s4_mtc_default.png'),
///     );
///   });
///
/// To regenerate goldens locally:
///   cd apps/mobile && flutter test --update-goldens test/goldens/
///
/// CI runs only the iphone14Pro subset (see ci.yml). Galaxy A14 goldens
/// exist for local manual runs by Julien per PERF-04 manual gate protocol.
library screen_pump;

import 'package:flutter/material.dart';
import 'package:flutter_test/flutter_test.dart';
import 'package:flutter_localizations/flutter_localizations.dart';
import 'package:mint/l10n/app_localizations.dart';

enum GoldenDevice {
  iphone14Pro,   // 390x844 @ 3.0x — CI device
  galaxyA14,     // ~411x914 logical @ 2.625x — local-only
}

class _DeviceSpec {
  const _DeviceSpec(this.size, this.devicePixelRatio);
  final Size size;
  final double devicePixelRatio;
}

const _kIphone14Pro = _DeviceSpec(Size(390, 844), 3.0);
const _kGalaxyA14 = _DeviceSpec(Size(411, 914), 2.625);

_DeviceSpec _specFor(GoldenDevice d) =>
    d == GoldenDevice.iphone14Pro ? _kIphone14Pro : _kGalaxyA14;

Future<void> pumpScreen(
  WidgetTester tester, {
  required Widget child,
  GoldenDevice device = GoldenDevice.iphone14Pro,
  Locale locale = const Locale('fr', 'CH'),
  bool disableAnimations = false,
}) async {
  final spec = _specFor(device);
  await tester.pumpWidget(
    MaterialApp(
      locale: locale,
      localizationsDelegates: const [
        AppLocalizations.delegate,
        GlobalMaterialLocalizations.delegate,
        GlobalWidgetsLocalizations.delegate,
        GlobalCupertinoLocalizations.delegate,
      ],
      supportedLocales: AppLocalizations.supportedLocales,
      home: MediaQuery(
        data: MediaQueryData(
          size: spec.size,
          devicePixelRatio: spec.devicePixelRatio,
          disableAnimations: disableAnimations,
        ),
        child: Material(child: child),
      ),
    ),
  );
  await tester.pumpAndSettle();
}
```

Create `test/goldens/helpers/screen_pump_test.dart` with the 3 unit tests above. Use `tester.binding.window` or `tester.view` (per Flutter version) to assert size + DPR after pumpScreen runs.
