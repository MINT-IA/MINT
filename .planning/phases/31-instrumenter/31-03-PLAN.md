    ```dart
    // apps/mobile/lib/widgets/mint_custom_paint_mask.dart
    //
    // Default-deny wrapper around CustomPaint for nLPD compliance.
    // sentry_flutter 9.14.0 privacy.maskAllText + maskAllImages do NOT cover
    // CustomPaint (canvas-rendered pixels). MINT uses CustomPaint for charts
    // rendering CHF / AVS / IBAN / ratios across Coach / Budget / Explorer.
    //
    // Rule (D-06): wrap EVERY financial-data CustomPaint in MintCustomPaintMask.
    // Only chrome (logos, dividers, spacers) may opt-out via SentryUnmask —
    // and every opt-out requires a commit message `unmask: <reason>`.
    //
    // Audit artefact: .planning/research/SENTRY_REPLAY_REDACTION_AUDIT.md

    import 'package:flutter/widgets.dart';
    import 'package:sentry_flutter/sentry_flutter.dart';

    /// Wrap a financial-data CustomPaint for Sentry Session Replay masking.
    ///
    /// Example:
    /// ```dart
    /// MintCustomPaintMask(
    ///   child: CustomPaint(painter: RetirementChartPainter(values)),
    /// )
    /// ```
    class MintCustomPaintMask extends StatelessWidget {
      const MintCustomPaintMask({super.key, required this.child});

      final Widget child;

      @override
      Widget build(BuildContext context) {
        // SentryMask marks the subtree as redacted in Session Replay capture.
        // Fallback: if SentryMask is a no-op in debug builds, that's OK —
        // masking matters at recorded frames (prod/staging).
        return SentryMask(child: child);
      }
    }
    ```
    NOTE on sentry_flutter 9.14.0 API: `SentryMask` widget exists per RESEARCH §Don't Hand-Roll row 5 (`[CITED: docs.sentry.io/platforms/dart/guides/flutter/session-replay/privacy/]`). If the exact class name differs in 9.14.0 (e.g., `SentryMaskWidget`), the executor resolves via `flutter doctor` + `grep -r "class SentryMask" apps/mobile/.dart_tool/` OR by reading `pubspec.lock` → sentry package source. Fallback if API missing: wrap in a container with `RenderObject.visitor` annotation per Sentry SDK issue tracker; document in SUMMARY.md if fallback used.

    #!/usr/bin/env bash
    # tools/simulator/pii_audit_screens.sh
    # OBS-06 PII audit — navigate to 5 sensitive screens, screenshot each.
    # Prereq: walker.sh --quick-screenshot has booted+installed staging build.
    # Visual mask verification (the actual audit) happens in Sentry UI REPLAY,
    # this script only captures the "did I reach the screen" proof state.
    set -euo pipefail

    DEVICE="${MINT_WALKER_DEVICE:-iPhone 17 Pro}"
    BUNDLE="com.mint.mintMobile"
    OUTDIR=".planning/walker/pii-audit-$(date +%Y-%m-%d-%H%M%S)"
    mkdir -p "$OUTDIR"

    # 5 sensitive screens — GoRouter deep links (resolved from apps/mobile/lib/app.dart routes):
    #   CoachChat        → /coach/chat
    #   DocumentScan     → /coach/scan       (best deep-link guess; adjust if diff)
    #   ExtractionReviewSheet → triggered by scan flow (not deep-link; after DocumentScan)
    #   Onboarding       → /onboarding
    #   Budget           → /explore/budget OR /home?tab=budget (check app.dart:177-261)

    SCREENS=(
      "coach-chat:/coach/chat"
      "document-scan:/coach/scan"
      "onboarding:/onboarding"
      "budget:/home?tab=budget"
    )

    for entry in "${SCREENS[@]}"; do
      NAME="${entry%%:*}"
      PATH_="${entry#*:}"
      echo "[pii-audit] navigating to ${NAME} (${PATH_})"
      # simctl deep-link via xcrun openurl
      timeout 15s xcrun simctl openurl "$DEVICE" "mint://${PATH_}" 2>/dev/null || \
        timeout 15s xcrun simctl openurl "$DEVICE" "https://mint.app${PATH_}" 2>/dev/null || \
        echo "[WARN] deep-link failed for ${NAME} — manual nav required in Sentry UI audit"
      sleep 3
      timeout 10s xcrun simctl io "$DEVICE" screenshot "${OUTDIR}/${NAME}.png"
    done

    # ExtractionReviewSheet requires document flow — can't deep-link. Flag for manual.
    echo "[pii-audit] ExtractionReviewSheet requires manual scan-triggered capture"
    echo "[pii-audit] screenshots saved to ${OUTDIR}"
    echo "[pii-audit] next step: open staging Sentry UI, filter replays env:staging"
    echo "[pii-audit] pause on CHF/IBAN/AVS frames, verify black box overlay visible"
    echo "[pii-audit] commit findings to .planning/research/SENTRY_REPLAY_REDACTION_AUDIT.md"
