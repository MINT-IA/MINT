#!/usr/bin/env python3
"""CTX-03 lint: assert 5 critical rules bracket TOP (first 40L) AND BOTTOM (last 40L) of CLAUDE.md (D-06).
Exit 0 = pass. Exit 1 = fail (stderr details)."""
import sys, argparse, re
from pathlib import Path

# D-06: the 5 critical rules (keyword per rule for grep-friendly detection)
RULE_KEYWORDS = {
    'banned_terms':      re.compile(r'\bbanned\s*terms\b', re.IGNORECASE),
    'accents':           re.compile(r'\baccent(s|é)?\b|100%\s*FR', re.IGNORECASE),
    'retirement':        re.compile(r'retirement|retraite.*app|retraite-first', re.IGNORECASE),
    'financial_core':    re.compile(r'financial_core|\bfinancial_core/\b', re.IGNORECASE),
    'i18n':              re.compile(r'\bi18n\b|AppLocalizations', re.IGNORECASE),
}

WINDOW_LINES = 40

def check_window(text: str, window: str, lines: list[str]) -> list[str]:
    """Returns list of rule_ids missing from the `lines` window."""
    missing = []
    block = '\n'.join(lines)
    for rule_id, pat in RULE_KEYWORDS.items():
        if not pat.search(block):
            missing.append(f'{rule_id} (in {window})')
    return missing

def main():
    ap = argparse.ArgumentParser(description='Assert 5 critical rules appear in TOP (first 40L) AND BOTTOM (last 40L) of CLAUDE.md (D-06)')
    ap.add_argument('--file', default='CLAUDE.md')
    args = ap.parse_args()
    p = Path(args.file)
    if not p.exists():
        print(f'bracket-lint: MISSING {args.file}', file=sys.stderr); return 1
    lines = p.read_text(encoding='utf-8').splitlines()
    if len(lines) < 2 * WINDOW_LINES:
        print(f'bracket-lint: FAIL — {args.file} has {len(lines)} lines; cannot bracket {WINDOW_LINES}+{WINDOW_LINES}', file=sys.stderr)
        return 1
    top, bottom = lines[:WINDOW_LINES], lines[-WINDOW_LINES:]
    problems = check_window('\n'.join(top), 'TOP', top) + check_window('\n'.join(bottom), 'BOTTOM', bottom)
    if problems:
        print(f'bracket-lint: FAIL — missing rules in CLAUDE.md brackets:', file=sys.stderr)
        for m in problems:
            print(f'  missing: {m}', file=sys.stderr)
        return 1
    print(f'bracket-lint: OK — 5/5 rules present in TOP and BOTTOM of {args.file}')
    return 0

if __name__ == '__main__':
    sys.exit(main())

#!/usr/bin/env python3
"""CTX-03 audit: detects duplication between CLAUDE.md §5-7 (now mostly docs/AGENTS/*.md)
and .claude/skills/mint-*/SKILL.md (D-08).
Approach: split each source into non-empty paragraphs, compute Jaccard 5-gram similarity pairwise,
flag pairs > threshold (default 0.6) as "likely duplicate" for manual review.
Writes markdown report to .planning/phases/30.6-context-sanity-advanced/audits/redundancy-audit.md."""
import re, sys, argparse
from pathlib import Path
from itertools import product

REPORT_PATH = Path('.planning/phases/30.6-context-sanity-advanced/audits/redundancy-audit.md')
CLAUDE_MD = Path('CLAUDE.md')
AGENTS_DIR = Path('docs/AGENTS')
SKILLS = [
    Path('.claude/skills/mint-swiss-compliance/SKILL.md'),
    Path('.claude/skills/mint-flutter-dev/SKILL.md'),
    Path('.claude/skills/mint-backend-dev/SKILL.md'),
]

def tokens(text: str) -> list[str]:
    return re.findall(r'\w+', text.lower())

def ngrams(toks: list[str], n: int = 5) -> set[tuple]:
    if len(toks) < n:
        return {tuple(toks)} if toks else set()
    return {tuple(toks[i:i+n]) for i in range(len(toks) - n + 1)}

def jaccard(a: set, b: set) -> float:
    if not a and not b:
        return 0.0
    return len(a & b) / len(a | b) if (a or b) else 0.0

def paragraphs(path: Path) -> list[tuple[int, str]]:
    """Returns [(starting_line, text)] for each non-empty paragraph."""
    if not path.exists():
        return []
    lines = path.read_text(encoding='utf-8').splitlines()
    out, cur_start, cur = [], None, []
    for i, line in enumerate(lines, start=1):
        if line.strip():
            if cur_start is None:
                cur_start = i
            cur.append(line)
        elif cur:
            out.append((cur_start, '\n'.join(cur)))
            cur_start, cur = None, []
    if cur:
        out.append((cur_start, '\n'.join(cur)))
    return out

def run(threshold: float, strict: bool):
    sources = {CLAUDE_MD: paragraphs(CLAUDE_MD)}
    for f in AGENTS_DIR.glob('*.md'):
        sources[f] = paragraphs(f)
    for s in SKILLS:
        sources[s] = paragraphs(s)
    primary = {k: v for k, v in sources.items() if k == CLAUDE_MD or str(k).startswith('docs/AGENTS/')}
    secondary = {k: v for k, v in sources.items() if str(k).startswith('.claude/skills/')}

    duplicates = []  # (primary_path, p_line, secondary_path, s_line, similarity)
    for (p_path, p_pars), (s_path, s_pars) in product(primary.items(), secondary.items()):
        for p_line, p_text in p_pars:
            p_grams = ngrams(tokens(p_text))
            if not p_grams:
                continue
            for s_line, s_text in s_pars:
                s_grams = ngrams(tokens(s_text))
                if not s_grams:
                    continue
                sim = jaccard(p_grams, s_grams)
                if sim >= threshold:
                    duplicates.append((str(p_path), p_line, str(s_path), s_line, sim))
    duplicates.sort(key=lambda d: -d[4])

    REPORT_PATH.parent.mkdir(parents=True, exist_ok=True)
    out = []
    out.append('# CLAUDE.md ↔ mint-* skills — redundancy audit (D-08)')
    out.append('')
    out.append(f'**Threshold:** Jaccard 5-gram ≥ {threshold}')
    out.append(f'**Candidates found:** {len(duplicates)}')
    out.append('')
    out.append('## Duplicates (ordered by similarity)')
    out.append('')
    out.append('| # | Primary (CLAUDE.md / AGENTS) | Secondary (skill) | Similarity | Proposed action |')
    out.append('|---|------------------------------|--------------------|-----------|------------------|')
    for i, (p_path, p_line, s_path, s_line, sim) in enumerate(duplicates[:50], 1):
        action = 'MOVE to skill (keep 1-liner in CLAUDE.md)' if sim > 0.8 else 'REVIEW manually'
        out.append(f'| {i} | {p_path}:{p_line} | {s_path}:{s_line} | {sim:.2f} | {action} |')
    if not duplicates:
        out.append('| — | none | — | — | audit CLEAN |')
    out.append('')
    out.append('## Decisions (manual review)')
    out.append('')
    out.append('Per D-08: CLAUDE.md keeps 1-line summary per skill ("Swiss compliance → `/mint-swiss-compliance`"), detail moves to skill file. Ré-architecture complète skills ↔ CLAUDE.md = deferred v2.9+.')
    out.append('')
    out.append('For each high-similarity duplicate (>0.8) above, Julien decides on PR review:')
    out.append('- [ ] Accept as-is (1-liner in CLAUDE.md is intentional redundancy)')
    out.append('- [ ] Drop from CLAUDE.md (content lives in skill)')
    out.append('- [ ] Drop from skill (content is more general, belongs in CLAUDE.md)')
    REPORT_PATH.write_text('\n'.join(out))

    if strict:
        high = [d for d in duplicates if d[4] > 0.8]
        if high:
            print(f'strict: FAIL — {len(high)} high-similarity (>0.8) duplicate(s) unresolved. See {REPORT_PATH}', file=sys.stderr)
            return 1
    print(f'audit: wrote {REPORT_PATH} ({len(duplicates)} candidates)')
    return 0
