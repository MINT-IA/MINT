    class CheckInAmountParser {
      CheckInAmountParser._();

      static const double _maxAmount = 999999.0;

      /// Extract first CHF amount from free text.
      /// Handles Swiss formatting: apostrophe thousands (1'500), comma decimal (1,50),
      /// space separator (1 500), "CHF" prefix/suffix.
      /// Returns null if no valid amount found or amount <= 0 or > 999999.
      static double? parseAmount(String text) {
        if (text.isEmpty) return null;
        final cleaned = text.replaceAll('\u00a0', ' ');
        final match = RegExp(r"\d[\d'\.\s]*(?:[.,]\d+)?").firstMatch(cleaned);
        if (match == null) return null;
        final raw = match.group(0)!
            .replaceAll("'", '')
            .replaceAll(' ', '')
            .replaceAll(',', '.');
        final value = double.tryParse(raw);
        if (value == null || value <= 0 || value > _maxAmount) return null;
        return value;
      }
    }
    ```

    /// [hasCheckedInThisMonth] — true if CoachProfile.checkIns contains current month.
    Future<void> scheduleCheckinReminder({required bool hasCheckedInThisMonth}) async {
      // Cancel any existing reminder first
      await _plugin.cancel(_idCheckinReminder5d);

      if (hasCheckedInThisMonth) return; // Already checked in — no reminder needed

    /// Streak at risk: either (a) engagement streak > 3 && missed yesterday,
    /// or (b) no monthly check-in for current month AND day >= 28 (2 days before month-end).
    static void _checkStreakAtRisk(
      List<JitaiNudge> nudges,
      int? currentStreak,
      bool? engagedYesterday,
      DateTime now,
      S? l, {
      CoachProfile? profile,  // NEW param — nullable for backward compat
    }) {
      // Original engagement streak logic
      if (currentStreak != null && engagedYesterday != null) {
        if (currentStreak > 3 && !engagedYesterday) {
          nudges.add(JitaiNudge(
            type: NudgeType.streakAtRisk,
            title: l?.nudgeStreakRiskTitle ?? 'Ta série est en danger\u00a0!',
            message: 'Tu as une série de $currentStreak\u00a0jours. '
                'Une petite action aujourd\'hui suffit pour la maintenir.',
            actionRoute: '/home?tab=1',
            actionLabel: l?.nudgeStreakRiskAction ?? 'Continuer ma série',
            triggeredAt: now,
            priority: NudgePriority.high,
          ));
          return; // Don't double-fire
        }
      }

      // Monthly check-in streak: day >= 28 and no check-in for current month
      if (profile != null && now.day >= 28) {
        final currentMonth = DateTime(now.year, now.month);
        final hasCheckInThisMonth = profile.checkIns.any(
          (c) => c.month.year == currentMonth.year && c.month.month == currentMonth.month,
        );
        if (!hasCheckInThisMonth) {
          final daysLeft = DateTime(now.year, now.month + 1, 0).day - now.day;
          nudges.add(JitaiNudge(
            type: NudgeType.streakAtRisk,
            title: l?.streakAtRiskTitle ?? 'Ta série est en jeu\u00a0!',
            message: l?.streakAtRiskBody?.call(daysLeft, profile.checkIns.length) ??
                'Il te reste $daysLeft jours pour maintenir ta série de ${profile.checkIns.length} mois.',
            actionRoute: '/home?tab=1&intent=monthlyCheckIn',
            actionLabel: l?.nudgeStreakRiskAction ?? 'Faire mon point du mois',
            triggeredAt: now,
            priority: NudgePriority.high,
          ));
        }
      }
    }
    ```
