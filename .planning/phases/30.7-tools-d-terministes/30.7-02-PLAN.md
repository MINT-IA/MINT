"""Phase 30.7 TOOL-03: validate_arb_parity.

Subprocess-wraps tools/checks/arb_parity.py (a Phase 34 GUARD-05 deliverable
that does not yet exist at Phase 30.7 ship time).

Graceful fallback: when the script is missing, return a payload with
``status="lint_not_available"`` that is DISTINCT from ``"ok"``. Agents must
not conflate the two — see RESEARCH.md Pitfall 5.
"""
from __future__ import annotations

import subprocess
import sys
from pathlib import Path
from typing import Literal, Optional

from pydantic import BaseModel, Field

TOOL_VERSION = "30.7.0"
SUBPROCESS_TIMEOUT_S = 30
MAX_OUTPUT_CHARS = 4_000

# Resolve relative to the repo root (this file lives at tools/mcp/mint-tools/tools/arb_parity.py,
# three parents up is the repo root).
_REPO_ROOT = Path(__file__).resolve().parents[3]
ARB_PARITY_SCRIPT = _REPO_ROOT / "tools" / "checks" / "arb_parity.py"


Status = Literal["ok", "drift_detected", "lint_not_available", "timeout", "error"]


class ArbParityResult(BaseModel):
    version: str = Field(default=TOOL_VERSION)
    status: Status
    exit_code: Optional[int] = None
    reason: str = ""
    stdout: str = ""
    stderr: str = ""
    script_expected_at: str = ""


def _truncate(s: str) -> str:
    """Keep the tail of long outputs — drift reports put the error at the bottom."""
    if s is None:
        return ""
    if len(s) <= MAX_OUTPUT_CHARS:
        return s
    return "...[truncated]...\n" + s[-MAX_OUTPUT_CHARS:]


def validate_arb_parity() -> ArbParityResult:
    """Run the ARB parity lint if present, otherwise return structured fallback.

    IMPORTANT for agents: if status == 'lint_not_available', ARB parity is
    NOT verified. Do not conclude parity is OK. Fallback status exists so the
    MCP tool surface stays stable even before Phase 34 ships arb_parity.py.
    """
    if not ARB_PARITY_SCRIPT.exists():
        return ArbParityResult(
            status="lint_not_available",
            reason="Phase 34 GUARD-05 (tools/checks/arb_parity.py) not yet shipped",
            script_expected_at=str(ARB_PARITY_SCRIPT),
        )

    try:
        proc = subprocess.run(
            [sys.executable, str(ARB_PARITY_SCRIPT)],
            capture_output=True,
            text=True,
            timeout=SUBPROCESS_TIMEOUT_S,
            cwd=str(_REPO_ROOT),
        )
    except subprocess.TimeoutExpired:
        return ArbParityResult(
            status="timeout",
            reason=f"subprocess exceeded {SUBPROCESS_TIMEOUT_S}s timeout",
            script_expected_at=str(ARB_PARITY_SCRIPT),
        )
    except (OSError, ValueError) as exc:
        return ArbParityResult(
            status="error",
            reason=f"subprocess failed: {exc}",
            script_expected_at=str(ARB_PARITY_SCRIPT),
        )

    status: Status = "ok" if proc.returncode == 0 else "drift_detected"
    return ArbParityResult(
        status=status,
        exit_code=proc.returncode,
        stdout=_truncate(proc.stdout or ""),
        stderr=_truncate(proc.stderr or ""),
        script_expected_at=str(ARB_PARITY_SCRIPT),
    )
```

"""Phase 30.7 TOOL-04: check_accent_patterns.

Wraps tools/checks/accent_lint_fr.scan_text (added in Wave 0 Task 2). Shares
the 14-pattern list via import — do NOT duplicate PATTERNS here. Phase 34 will
extend the list; this tool inherits the extension for free.
"""
from __future__ import annotations

import sys
from pathlib import Path
from typing import Optional

from pydantic import BaseModel, Field

TOOL_VERSION = "30.7.0"
MAX_TEXT_LEN = 10_000

# Resolve repo root and put it on sys.path so `from tools.checks.accent_lint_fr import ...` works
# when the MCP server is spawned from the repo root (Wave 2 server.py also does this).
_REPO_ROOT = Path(__file__).resolve().parents[3]
if str(_REPO_ROOT) not in sys.path:
    sys.path.insert(0, str(_REPO_ROOT))

from tools.checks.accent_lint_fr import scan_text as _scan_text  # noqa: E402


class AccentHit(BaseModel):
    line: int
    snippet: str
    pattern: str
    suggestion: str


class AccentResult(BaseModel):
    version: str = Field(default=TOOL_VERSION)
    clean: bool
    violations: list[AccentHit]
    text_length: int


def _parse_correction(raw: str) -> tuple[str, str]:
    """Split the 'pattern -> correction' string produced by accent_lint_fr."""
    if " -> " not in raw:
        return raw, ""
    pat, _, correction = raw.partition(" -> ")
    return pat, correction


def check_accent_patterns(text: Optional[str]) -> AccentResult:
    """Scan French text for ASCII-flattened accents (14 patterns from Phase 30.5 baseline).

    Inputs longer than MAX_TEXT_LEN are truncated at the scan layer but
    text_length reports the original length.
    """
    original_len = len(text) if text else 0
    if not text:
        return AccentResult(clean=True, violations=[], text_length=0)

    scanned = text[:MAX_TEXT_LEN]
    raw_hits = _scan_text(scanned)
    violations: list[AccentHit] = []
    for lineno, snippet, raw_pattern in raw_hits:
        pattern, suggestion = _parse_correction(raw_pattern)
        violations.append(
            AccentHit(
                line=lineno,
                snippet=snippet,
                pattern=pattern,
                suggestion=suggestion,
            )
        )

    return AccentResult(
        clean=(not violations),
        violations=violations,
        text_length=original_len,
    )
```
