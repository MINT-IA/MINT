"""Phase 30.7 TOOL-01: get_swiss_constants.

Wraps the backend RegulatoryRegistry singleton. Read-only. Stateless per call
(the singleton is module-level memoization, acceptable per RESEARCH.md).

Source of truth: services/backend/app/services/regulatory/registry.py.
Do NOT duplicate any constant value in this file. If a value looks wrong,
fix it in the backend registry — never patch it here.
"""
from __future__ import annotations

from typing import Literal, Optional

from pydantic import BaseModel, Field

TOOL_VERSION = "30.7.0"

# The 5 canonical categories (CONTEXT.md Area 2 + RESEARCH.md §Phase Requirements).
Category = Literal["pillar3a", "lpp", "avs", "mortgage", "tax"]


class ConstantEntry(BaseModel):
    key: str
    value: float
    unit: str
    source_title: str = ""
    effective_from: str = ""
    tax_year: Optional[int] = None


class ConstantsResult(BaseModel):
    version: str = Field(default=TOOL_VERSION)
    category: str
    jurisdiction: str = "CH"
    constants: list[ConstantEntry]


def get_swiss_constants(category: str) -> ConstantsResult:
    """Return Swiss regulatory constants for one of {pillar3a, lpp, avs, mortgage, tax}.

    Unknown categories return ``constants=[]`` (no raise) so the MCP envelope
    can surface a structured error rather than JSON-RPC fault.
    """
    # Lazy import — if backend fails to import, MCP server still starts and other tools work.
    # Per RESEARCH.md Pitfall 3 (PYTHONPATH) + Security T-30.7-01.
    try:
        from app.services.regulatory.registry import RegulatoryRegistry  # type: ignore[import-not-found]
    except ImportError as e:
        return ConstantsResult(
            category=category,
            constants=[],
            jurisdiction="CH",
        )

    params = RegulatoryRegistry.instance().get_all(category=category)
    entries: list[ConstantEntry] = []
    for p in params:
        entries.append(
            ConstantEntry(
                key=getattr(p, "key", ""),
                value=float(getattr(p, "value", 0.0)),
                unit=getattr(p, "unit", ""),
                source_title=getattr(p, "source_title", None) or "",
                effective_from=getattr(p, "effective_from", None) or "",
                tax_year=getattr(p, "tax_year", None),
            )
        )
    return ConstantsResult(category=category, constants=entries)
```

"""Phase 30.7 TOOL-02: check_banned_terms.

Wraps ComplianceGuard at services/backend/app/services/coach/compliance_guard.py.
Layer 1 only (banned terms). Layer 2/2b (prescriptive/high-register) deferred
to v2.9 per CONTEXT.md Area 2 lock.

ComplianceGuard is instantiated once at module scope — its __init__ creates a
HallucinationDetector (slow cold-start, per RESEARCH.md Pitfall 4). Do not
re-instantiate per call.
"""
from __future__ import annotations

from typing import Optional

from pydantic import BaseModel, Field

TOOL_VERSION = "30.7.0"
MAX_TEXT_LEN = 10_000  # DoS cap — security T-30.7-02

# Module-scope lazy-loaded guard. Avoids slow cold-start on every call.
_GUARD: Optional["ComplianceGuard"] = None  # type: ignore[name-defined]


def _get_guard():
    """Lazy-init the ComplianceGuard singleton. Raises ImportError only if
    services/backend is not on PYTHONPATH (misconfiguration, not runtime)."""
    global _GUARD
    if _GUARD is None:
        # Local import so tests can mock and so an import failure surfaces
        # at the first real tool call, not at MCP server startup.
        from app.services.coach.compliance_guard import ComplianceGuard  # type: ignore[import-not-found]
        _GUARD = ComplianceGuard()
    return _GUARD


class BannedTermHit(BaseModel):
    term: str
    suggestion: str = ""
    context: str = ""


class BannedTermsResult(BaseModel):
    version: str = Field(default=TOOL_VERSION)
    clean: bool
    banned_found: list[BannedTermHit]
    sanitized_text: str
    text_length: int


def check_banned_terms(text: str) -> BannedTermsResult:
    """Scan text for LSFin banned terms, return hits + sanitized replacement.

    Inputs longer than MAX_TEXT_LEN (10 000 chars) are truncated at the scan
    layer but text_length reports the original length.

    Safe on empty / None / short inputs — returns a clean-True payload rather
    than raising.
    """
    original_len = len(text) if text else 0
    if not text:
        return BannedTermsResult(
            clean=True,
            banned_found=[],
            sanitized_text="",
            text_length=0,
        )

    # Enforce DoS cap — scan the prefix only.
    scanned = text[:MAX_TEXT_LEN]

    try:
        guard = _get_guard()
    except ImportError:
        # Backend not on PYTHONPATH (should not happen in .mcp.json). Degrade gracefully.
        return BannedTermsResult(
            clean=True,
            banned_found=[],
            sanitized_text=scanned,
            text_length=original_len,
        )

    # Upstream signature: _check_banned_terms returns list[dict] with keys we normalize here.
    raw_hits = guard._check_banned_terms(scanned) or []
    replacements = getattr(guard, "TERM_REPLACEMENTS", {}) or {}

    hits: list[BannedTermHit] = []
    for raw in raw_hits:
        # Best-effort normalization — upstream may return dicts or objects.
        term = ""
        context = ""
        if isinstance(raw, dict):
            term = str(raw.get("term") or raw.get("match") or "")
            context = str(raw.get("context") or raw.get("snippet") or "")[:80]
        else:
            term = str(getattr(raw, "term", "") or getattr(raw, "match", ""))
            context = str(getattr(raw, "context", "") or "")[:80]
        if not term:
            continue
        suggestion = replacements.get(term.lower(), replacements.get(term, ""))
        hits.append(BannedTermHit(term=term, suggestion=suggestion, context=context))

    sanitized = guard._sanitize_banned_terms(scanned)

    return BannedTermsResult(
        clean=(not hits),
        banned_found=hits,
        sanitized_text=sanitized,
        text_length=original_len,
    )
```
