#!/usr/bin/env python3
"""CTX-01: rewrite MEMORY.md links from `](feedback_xxx.md)` to `](topics/feedback_xxx.md)`.
Idempotent. Backup MEMORY.md.bak before first write.
D-03: topics is flat (no feedback/ project/ reference/ subdirs)."""
import re, sys, argparse, shutil
from pathlib import Path

MEMORY_DIR = Path.home() / '.claude' / 'projects' / '-Users-julienbattaglia-Desktop-MINT' / 'memory'
MEMORY_MD = MEMORY_DIR / 'MEMORY.md'
# Matches `](name.md)` where name starts with feedback_|project_|reference_|user_ or other custom prefix (not already prefixed with topics/ or archive/)
LINK_PATTERN = re.compile(r'\]\((?!https?://|/|topics/|archive/|\#)([A-Za-z0-9_\-]+\.md)\)')

def main():
    p = argparse.ArgumentParser(description='Rewrite MEMORY.md links to topics/')
    p.add_argument('--dry-run', action='store_true')
    args = p.parse_args()
    if not MEMORY_MD.exists():
        print(f'MEMORY.md missing at {MEMORY_MD}', file=sys.stderr)
        return 1
    text = MEMORY_MD.read_text(encoding='utf-8')
    rewrites = 0
    def repl(m):
        nonlocal rewrites
        rewrites += 1
        return f']({"topics/"}{m.group(1)})'
    new_text = LINK_PATTERN.sub(repl, text)
    if args.dry_run:
        print(f'{rewrites} links would be rewritten')
        return 0
    if new_text != text:
        bak = MEMORY_MD.with_suffix('.md.bak')
        if not bak.exists():
            shutil.copy2(MEMORY_MD, bak)
        MEMORY_MD.write_text(new_text, encoding='utf-8')
        print(f'rewrote {rewrites} links in {MEMORY_MD}')
    else:
        print('no changes (already rewritten or no links matched)')
    return 0

if __name__ == '__main__':
    sys.exit(main())
```

#!/usr/bin/env python3
"""CTX-01 30-day GC: archives memory/topics/*.md with mtime >30d to memory/archive/YYYY-MM/.
D-02: '30j retention, 0 truncation silencieuse' — the real gate.
D-03: flat topics/, archive by YYYY-MM from file mtime (NOT frontmatter).

If Plan 00 A4 spike returned FAIL (identical mtimes), this script includes a fallback:
- primary: use file mtime via Path.stat().st_mtime
- fallback: if mtime identical across a sample, use git log --diff-filter=A to extract birthdate"""
import os, sys, argparse, shutil, time
from pathlib import Path
from datetime import datetime, timezone

MEMORY_DIR = Path.home() / '.claude' / 'projects' / '-Users-julienbattaglia-Desktop-MINT' / 'memory'
TOPICS_DIR = MEMORY_DIR / 'topics'
ARCHIVE_DIR = MEMORY_DIR / 'archive'
RETENTION_DAYS = 30
NOW = time.time()

def file_age_seconds(p: Path) -> float:
    try:
        return NOW - p.stat().st_mtime
    except OSError:
        return 0

def archive_bucket(p: Path) -> Path:
    """YYYY-MM bucket based on file mtime."""
    mtime = p.stat().st_mtime
    dt = datetime.fromtimestamp(mtime, tz=timezone.utc)
    return ARCHIVE_DIR / f'{dt.year:04d}-{dt.month:02d}'

def main():
    ap = argparse.ArgumentParser(description='30d retention GC for memory/topics/')
    ap.add_argument('--dry-run', action='store_true')
    ap.add_argument('--days', type=int, default=RETENTION_DAYS)
    args = ap.parse_args()
    if not TOPICS_DIR.exists():
        print(f'topics/ missing at {TOPICS_DIR} — run migration first', file=sys.stderr)
        return 1
    threshold = args.days * 86400
    moved = []
    for p in sorted(TOPICS_DIR.glob('*.md')):
        # Skip lefthook self-test fixtures (managed by tools/checks/lefthook_self_test.sh EXIT trap)
        if p.name.startswith('__LEFTHOOK_SELF_TEST_'):
            continue
        age = file_age_seconds(p)
        if age <= threshold:
            continue
        bucket = archive_bucket(p)
        dest = bucket / p.name
        if args.dry_run:
            moved.append((str(p), str(dest), age / 86400))
            continue
        bucket.mkdir(parents=True, exist_ok=True)
        if dest.exists():
            # Already archived by a previous run; remove duplicate at source to stay idempotent
            if dest.stat().st_size == p.stat().st_size:
                p.unlink()
            else:
                # Size differs — keep both, rename new copy with suffix
                dest = dest.with_name(p.stem + f'.{int(p.stat().st_mtime)}.md')
                shutil.move(str(p), str(dest))
        else:
            shutil.move(str(p), str(dest))
        moved.append((str(p), str(dest), age / 86400))
    if args.dry_run:
        for src, dst, days in moved:
            print(f'DRY {src} → {dst} (age={days:.1f}d)')
        print(f'DRY total: {len(moved)}')
    else:
        for src, dst, days in moved:
            print(f'archived {src} → {dst} (age={days:.1f}d)')
        print(f'total: {len(moved)}')
    return 0

if __name__ == '__main__':
    sys.exit(main())
```

#!/usr/bin/env python3
"""CTX-01 retention gate invoked by lefthook pre-commit (D-04 scope: ONLY this lint).
Exit 0 = pass. Exit 1 = fail (loud stderr).
Gate criteria (D-02):
  HARD: every memory/topics/*.md with mtime >30d MUST also exist in memory/archive/YYYY-MM/ OR be gone
  SOFT: MEMORY.md <100 lines (warning to stderr, does not fail)"""
import os, sys, argparse, time
from pathlib import Path
from datetime import datetime, timezone

MEMORY_DIR = Path.home() / '.claude' / 'projects' / '-Users-julienbattaglia-Desktop-MINT' / 'memory'
TOPICS_DIR = MEMORY_DIR / 'topics'
MEMORY_MD = MEMORY_DIR / 'MEMORY.md'
RETENTION_DAYS = 30
INDEX_WARN_LINES = 100

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument('--strict', action='store_true', help='Fail on retention breach (default behavior)')
    args = ap.parse_args()
    if not TOPICS_DIR.exists():
        print(f'retention: topics/ missing — run migration (Plan 02 Task 1)', file=sys.stderr)
        return 1
    now = time.time()
    threshold = RETENTION_DAYS * 86400
    breached = []
    for p in TOPICS_DIR.glob('*.md'):
        # Skip lefthook self-test fixtures (namespaced, owned by tools/checks/lefthook_self_test.sh)
        if p.name.startswith('__LEFTHOOK_SELF_TEST_'):
            continue
        age = now - p.stat().st_mtime
        if age > threshold:
            breached.append((str(p), age / 86400))
    # SOFT warning on INDEX size
    if MEMORY_MD.exists():
        lines = len(MEMORY_MD.read_text(encoding='utf-8').splitlines())
        if lines > INDEX_WARN_LINES:
            print(f'retention WARNING: {MEMORY_MD} has {lines} lines (target <{INDEX_WARN_LINES}). This is a warning per D-02, not a failure.', file=sys.stderr)
    if breached:
        print(f'retention: FAIL — {len(breached)} file(s) in topics/ have mtime >30d and not archived:', file=sys.stderr)
        for p, days in breached:
            print(f'  {p} ({days:.1f}d old)', file=sys.stderr)
        print(f'  → Run: python3 tools/memory/gc.py', file=sys.stderr)
        return 1
    print('retention: OK — 30j gate green')
    return 0

if __name__ == '__main__':
    sys.exit(main())
