```python
#!/usr/bin/env python3
"""CTX-02 metric (a): parse `git log --author='Co-Authored-By: Claude'` last 7d,
run accent_lint_fr + no_hardcoded_fr on each commit's diff, populate commits + violations tables."""
import subprocess, sqlite3, sys
from pathlib import Path
from datetime import datetime, timedelta

DB_PATH = Path('.planning/agent-drift/drift.db')
LINTS = [
    ('accent_lint_fr', 'tools/checks/accent_lint_fr.py'),
    ('no_hardcoded_fr', 'tools/checks/no_hardcoded_fr.py'),
]

def parse_git_log(days=7):
    """Returns list of (sha, author_classified, committed_at_unix, subject)."""
    since = (datetime.utcnow() - timedelta(days=days)).isoformat()
    # Full log with body so we can detect Co-Authored-By
    result = subprocess.run(
        ['git', 'log', f'--since={since}', '--format=%H%x1f%ct%x1f%s%x1f%b%x1e'],
        capture_output=True, text=True, check=True
    )
    commits = []
    for entry in result.stdout.split('\x1e'):
        if not entry.strip():
            continue
        parts = entry.strip().split('\x1f')
        if len(parts) < 3:
            continue
        sha, ct, subject = parts[0], int(parts[1]), parts[2]
        body = parts[3] if len(parts) > 3 else ''
        is_claude = 'Co-Authored-By: Claude' in body or 'Co-Authored-By: claude' in body.lower()
        author = 'claude-agent' if is_claude else 'human'
        commits.append((sha, author, ct, subject))
    return commits

def files_changed(sha):
    """Returns list of files modified in commit `sha`."""
    r = subprocess.run(['git', 'show', '--name-only', '--format=', sha], capture_output=True, text=True, check=True)
    return [f for f in r.stdout.splitlines() if f.strip()]

def run_lint_on_file(lint_path, target_file):
    """Returns list of (line_number, snippet) violations."""
    if not Path(target_file).exists():
        return []
    r = subprocess.run(['python3', lint_path, '--file', target_file], capture_output=True, text=True)
    if r.returncode == 0:
        return []
    # Parse stderr for "path:line: snippet" (our lint convention)
    violations = []
    for line in r.stderr.splitlines():
        if ':' in line:
            try:
                # "path:12: snippet text"
                _, lineno, snippet = line.split(':', 2)
                violations.append((int(lineno.strip()), snippet.strip()))
            except (ValueError, IndexError):
                continue
    return violations

def main(days=7, db_path=DB_PATH):
    db_path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(db_path)
    for sha, author, ct, subject in parse_git_log(days):
        conn.execute('INSERT OR REPLACE INTO commits (sha, author, committed_at, subject) VALUES (?, ?, ?, ?)',
                     (sha, author, ct, subject))
        if author != 'claude-agent':
            continue
        for f in files_changed(sha):
            for lint_name, lint_path in LINTS:
                for lineno, snippet in run_lint_on_file(lint_path, f):
                    conn.execute('INSERT INTO violations (sha, lint, file_path, line_number, snippet, detected_at) VALUES (?, ?, ?, ?, ?, ?)',
                                 (sha, lint_name, f, lineno, snippet[:200], int(datetime.utcnow().timestamp())))
    conn.commit()
    conn.close()

if __name__ == '__main__':
    main()
```

```python
#!/usr/bin/env python3
"""Early-ship accent lint for FR missing diacritics (CTX-02 metric a ingestion).
Phase 34 GUARD-04 will add more patterns + CI integration.
Pattern list from CONTEXT.md and MEMORY.md historical violations.
Exit 0 = clean; Exit 1 = violations found (prints to stderr for ingest_git.py to parse)."""
import re, sys, argparse
from pathlib import Path

# Each pattern = (ascii_bad_pattern, correct_example)
# Use word boundaries to avoid false positives on valid tokens
PATTERNS = [
    (r'\bcreer\b', 'créer'),
    (r'\bdecouvrir\b', 'découvrir'),
    (r'\beclairage\b', 'éclairage'),
    (r'\bsecurite\b', 'sécurité'),
    (r'\bliberer\b', 'libérer'),
    (r'\bpreter\b', 'prêter'),
    (r'\brealiser\b', 'réaliser'),
    (r'\bdeja\b', 'déjà'),
    (r'\brecu\b', 'reçu'),
    (r'\belaborer\b', 'élaborer'),
    (r'\bregler\b', 'régler'),
    (r'\bspecialistes?\b', 'spécialiste(s)'),
    (r'\bretraite[^\w]', 'retraité / retraite — contexte'),  # retraite itself OK but often accent needed nearby
]

TEXT_EXTS = {'.dart', '.py', '.arb', '.md'}
EXCLUDE_DIRS = {'.git', 'node_modules', '.dart_tool', 'build', '__pycache__', 'docs/archive', '.planning/archives'}

def scan_file(path: Path):
    """Returns list of (lineno, snippet, pattern) violations."""
    try:
        text = path.read_text(encoding='utf-8', errors='ignore')
    except OSError:
        return []
    violations = []
    for lineno, line in enumerate(text.splitlines(), start=1):
        for pat, correct in PATTERNS:
            if re.search(pat, line, re.IGNORECASE):
                violations.append((lineno, line.strip()[:140], f'{pat} → suggest {correct}'))
    return violations

def main():
    p = argparse.ArgumentParser()
    p.add_argument('--file', help='Lint a single file')
    p.add_argument('--scope', nargs='*', default=['apps/mobile/lib', 'services/backend/app', 'tools'])
    args = p.parse_args()
    if args.file:
        paths = [Path(args.file)]
    else:
        paths = []
        for s in args.scope:
            root = Path(s)
            if not root.exists(): continue
            paths.extend([p for p in root.rglob('*') if p.is_file() and p.suffix in TEXT_EXTS])
    found = 0
    for path in paths:
        rel = str(path)
        if any(ex in rel for ex in EXCLUDE_DIRS): continue
        for lineno, snippet, pat in scan_file(path):
            print(f'{rel}:{lineno}: {snippet} ({pat})', file=sys.stderr)
            found += 1
    return 1 if found else 0

if __name__ == '__main__':
```

```python
#!/usr/bin/env python3
"""Render nightly markdown report with 4 metrics from drift.db (CTX-02)."""
import sqlite3, sys, argparse
from datetime import datetime
from pathlib import Path

DB_PATH = Path('.planning/agent-drift/drift.db')

METRIC_A_SQL = """
SELECT COUNT(DISTINCT c.sha) FILTER (WHERE v.sha IS NOT NULL) * 1.0 /
       NULLIF(COUNT(DISTINCT c.sha), 0) AS drift_rate
FROM commits c
LEFT JOIN violations v ON v.sha = c.sha
WHERE c.author = 'claude-agent'
  AND c.committed_at >= strftime('%s', 'now', '-7 days')
"""

METRIC_B_SQL = """
SELECT COUNT(DISTINCT s.session_id) FILTER (WHERE h.tool_use_index = 0) * 1.0 /
       NULLIF(COUNT(DISTINCT s.session_id), 0) AS context_hit_rate
FROM sessions s
LEFT JOIN context_hits h ON h.session_id = s.session_id
WHERE s.started_at >= strftime('%s', 'now', '-7 days')
"""

METRIC_C_SQL = """
SELECT AVG(total_input_tokens + total_output_tokens + total_cache_creation + total_cache_read)
FROM sessions
WHERE started_at >= strftime('%s', 'now', '-7 days')
"""

METRIC_D_SQL = """
SELECT ROUND(AVG(turns_to_correct), 2)
FROM golden_runs
WHERE run_at = (SELECT MAX(run_at) FROM golden_runs)
"""

def fetchone(conn, sql):
    cur = conn.execute(sql)
    row = cur.fetchone()
    return row[0] if row and row[0] is not None else None

def render(db_path=DB_PATH, out_path=None):
    if not db_path.exists():
        raise RuntimeError(f"drift.db not found at {db_path}. Run `dashboard.py init` first.")
    conn = sqlite3.connect(db_path)
    m_a = fetchone(conn, METRIC_A_SQL)
    m_b = fetchone(conn, METRIC_B_SQL)
    m_c = fetchone(conn, METRIC_C_SQL)
    m_d = fetchone(conn, METRIC_D_SQL)
    conn.close()

    today = datetime.utcnow().date().isoformat()
    lines = [
        f'# Agent drift report — {today}',
        '',
        '## Metrics (last 7d)',
        '',
        f'### Drift rate (metric a)',
        f'- **Value:** {f"{m_a*100:.1f}%" if m_a is not None else "n/a"}',
        f'- **Definition:** % commits by Co-Authored-By: Claude with ≥1 lint violation',
        f'- **Source:** `git log --author=\'Co-Authored-By: Claude\'` × `tools/checks/{{accent_lint_fr,no_hardcoded_fr}}.py`',
        '',
        f'### Context hit rate (metric b)',
        f'- **Value:** {f"{m_b*100:.1f}%" if m_b is not None else "n/a"}',
        f'- **Definition:** % sessions where gsd-prompt-guard.js detected ≥1 rule violation at first tool_use',
        f'- **Source:** `.claude/hooks/gsd-prompt-guard.js` → `.planning/agent-drift/context_hits.jsonl`',
        '',
        f'### Token cost per session (metric c)',
        f'- **Value:** {int(m_c) if m_c is not None else "n/a"} tokens (avg input+output+cache)',
        f'- **Definition:** mean total tokens per session from Claude Code JSONL transcripts',
        f'- **Source:** `.claude/projects/-Users-julienbattaglia-Desktop-MINT/*.jsonl` usage sums',
        '',
        f'### Time-to-first-correct-output (metric d)',
        f'- **Value:** {m_d if m_d is not None else "n/a"} turns avg (latest golden run)',
        f'- **Definition:** # turns to produce lint-clean output on 20 golden prompts',
        f'- **Source:** `tools/agent-drift/golden/run.sh` → golden_runs table',
        '',
    ]
    content = '\n'.join(lines)
    if out_path:
        Path(out_path).parent.mkdir(parents=True, exist_ok=True)
        Path(out_path).write_text(content)
    return content

def main():
    p = argparse.ArgumentParser()
    p.add_argument('--out', required=False)
    args = p.parse_args()
    out = args.out or f'.planning/agent-drift/{datetime.utcnow().date().isoformat()}.md'
    render(out_path=out)
    print(f'wrote {out}')
    return 0

if __name__ == '__main__':
    sys.exit(main())
```
