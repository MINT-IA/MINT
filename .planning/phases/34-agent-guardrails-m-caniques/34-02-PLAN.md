def get_added_lines(file_path: str) -> list[tuple[int, str]]:
    """Return (new_line_number, content) for lines ADDED in the staged diff."""
    result = subprocess.run(
        ['git', 'diff', '--staged', '--unified=0', '--no-renames',
         '--diff-filter=AM', '--', file_path],
        capture_output=True, text=True, check=False,
    )
    added: list[tuple[int, str]] = []
    cur_new: Optional[int] = None
    HUNK = re.compile(r'^@@ -\d+(?:,\d+)? \+(\d+)(?:,\d+)? @@')
    for line in result.stdout.splitlines():
        m = HUNK.match(line)
        if m:
            cur_new = int(m.group(1))
            continue
        if cur_new is None:
            continue
        if line.startswith('+++') or line.startswith('---'):
            continue

            added.append((cur_new, line[1:]))
            cur_new += 1
        elif line.startswith('-'):

  ```python
  #!/usr/bin/env python3
  """GUARD-02 — refuse added lines containing bare `catch (e) {}` Dart /
  `except Exception:` Python with no log/rethrow.

  Per CONTEXT 34-CONTEXT.md D-07: scans ONLY the added lines of the staged
  diff, not the full file content. This decouples Phase 34 from FIX-05
  migration of 388 existing bare-catches (Phase 36 scope).

  Per D-06: exempts test/integration_test/tests paths, Dart `async *`
  generators, and inline override `// lefthook-allow:bare-catch: <reason>`
  (reason must be ≥3 words). The override is accepted on the SAME line as
  the bare-catch OR on the IMMEDIATELY PRECEDING line — mirroring the
  Plan 34-03 `no_hardcoded_fr.py` semantics so reviewers and IDE tooling
  can treat the convention uniformly.

  Exit codes:
    0 — no bare-catch introduced by this diff
    1 — violation(s) found (stderr has `[no_bare_catch] path:line: snippet` rows)

  Technical English diagnostics — dev-facing, no ARB i18n (per M-1 carve-out,
  Phase 32-03 admin precedent). RESEARCH §Pitfall 8 self-compliance.
  """
  from __future__ import annotations
  import argparse
  import re
  import subprocess
  import sys
  from pathlib import Path
  from typing import List, Optional, Tuple

  # ─── Patterns (RESEARCH §Pattern 3) ───────────────────────────────────────
  DART_BARE_CATCH = [
      re.compile(r'}\s*catch\s*\(\s*(?:e|_|err|error)\s*\)\s*\{\s*\}'),
      re.compile(r'on\s+\w+\s+catch\s*\(\s*(?:e|_|err)\s*\)\s*\{\s*\}'),
  ]
  DART_LOG_TOKENS = (
      'Sentry.captureException', 'Sentry.captureMessage', 'SentryBreadcrumb',
      'debugPrint(', 'print(',
      'log(', 'logger.',
      'rethrow', 'throw',
      'FirebaseCrashlytics',
  )

  PY_BARE_EXCEPT = [
      re.compile(r'^\s*except\s*:\s*$'),
      re.compile(r'^\s*except\s+Exception\s*:\s*$'),
      re.compile(r'^\s*except\s+BaseException\s*:\s*$'),
  ]
  PY_LOG_TOKENS = (
      'logger.', 'logging.', 'log.',
      'sentry_sdk.capture', 'sentry_sdk.push_scope',
      'raise', 'print(',
  )

  # Override markers — accepted on the same line as the bare-catch OR on the
  # immediately preceding line. Reason length is enforced in
  # `_has_valid_override` (≥3 whitespace-separated words per D-06).
  OVERRIDE_DART = re.compile(r'//\s*lefthook-allow:\s*bare-catch:\s*(.+)')
  OVERRIDE_PY   = re.compile(r'#\s*lefthook-allow:\s*bare-catch:\s*(.+)')

  # D-06 authorises EXACTLY these four prefixes. Do NOT add a broad `tests/`
  # entry — it would exempt any future top-level `tests/` directory beyond
  # scope (W1 guard).
  EXEMPT_PATH_PREFIXES = (
      'apps/mobile/test/',
      'apps/mobile/integration_test/',
      'services/backend/tests/',
      'tests/checks/fixtures/',
  )


  # ─── Diff parser (RESEARCH §Pattern 2, VERIFIED /tmp/gittest 2026-04-22) ──
  def get_added_lines(file_path: str, repo_root: Optional[Path] = None) -> List[Tuple[int, str]]:
      """Return (new_line_number, content) for lines ADDED in the staged diff.

      Pure read-only — no writes to .git/index.lock (RESEARCH §Pattern 6).
      """
      cmd = [
          'git', 'diff', '--staged', '--unified=0', '--no-renames',
          '--diff-filter=AM', '--', file_path,
      ]
      kwargs = {'capture_output': True, 'text': True, 'check': False}
      if repo_root is not None:
          kwargs['cwd'] = str(repo_root)
      result = subprocess.run(cmd, **kwargs)
      added: List[Tuple[int, str]] = []
      cur_new: Optional[int] = None
      HUNK = re.compile(r'^@@ -\d+(?:,\d+)? \+(\d+)(?:,\d+)? @@')
      for line in result.stdout.splitlines():
          m = HUNK.match(line)
          if m:
              cur_new = int(m.group(1))
              continue
          if cur_new is None:
              continue
          if line.startswith('+++') or line.startswith('---'):
              continue
          if line.startswith('+'):
              added.append((cur_new, line[1:]))
              cur_new += 1
          elif line.startswith('-'):
              continue
          elif line.startswith('\\'):
              continue
          else:
              cur_new += 1
      return added


  def is_exempt_path(file_path: str) -> bool:
      normalized = file_path.replace('\\', '/').lstrip('./')
      return any(normalized.startswith(prefix) for prefix in EXEMPT_PATH_PREFIXES)


  def is_in_async_star(full_text: str, line_no: int) -> bool:
      """Return True if line_no falls within a Dart `async *` generator body.

      Heuristic (D-06): within 10 lines BEFORE line_no, find `async *` or
      `async*` on a line that also contains a function signature / arrow.
      """
      lines = full_text.splitlines()
      start = max(0, line_no - 10)
      for candidate in lines[start:line_no]:
          if re.search(r'\basync\s*\*', candidate):
              return True
      return False


  def has_surrounding_log_tokens(full_text: str, line_no: int, tokens: Tuple[str, ...]) -> bool:
      """Check if any log/raise token appears within the 5 lines following line_no."""
      lines = full_text.splitlines()
      # 1-indexed input, 0-indexed list
      start = max(0, line_no - 1)
      end = min(len(lines), line_no + 5)
      window = '\n'.join(lines[start:end])
      return any(tok in window for tok in tokens)


  def _has_valid_override(line: str, is_python: bool) -> bool:
      """Return True iff `line` carries a `lefthook-allow:bare-catch:` marker
      whose reason is ≥3 whitespace-separated words (D-06)."""
      pat = OVERRIDE_PY if is_python else OVERRIDE_DART
      m = pat.search(line)
      if not m:
          return False
      reason = m.group(1).strip()
      return len(reason.split()) >= 3


  def _override_in_preceding(lines: List[str], added_line_no: int, is_python: bool) -> bool:
      """Check the immediately preceding line (1-indexed added_line_no - 1 in file)
      for a valid `lefthook-allow:bare-catch:` override.

      `added_line_no` is the 1-indexed new-file line number produced by the
      git unified diff parser. `lines` is the full staged-file content split
      into 0-indexed lines (via `str.splitlines()`).
      """
      # If the bare-catch is on the first line, there is no preceding line.
      if added_line_no < 2:
          return False
      # Convert 1-indexed -> 0-indexed, then step back one more for preceding.
      idx_prev = added_line_no - 2
      if idx_prev < 0 or idx_prev >= len(lines):
          return False
      prev = lines[idx_prev]
      return _has_valid_override(prev, is_python=is_python)


  def scan_dart_added(
      added: List[Tuple[int, str]],
      full_text: str,
      file_path: str,
  ) -> List[str]:
      violations: List[str] = []
      lines = full_text.splitlines()
      for line_no, content in added:
          # Accept override on same line OR on the immediately preceding line.
          if _has_valid_override(content, is_python=False) or _override_in_preceding(lines, line_no, is_python=False):
              continue
          if is_in_async_star(full_text, line_no):
              continue
          for pat in DART_BARE_CATCH:
              if pat.search(content):
                  if has_surrounding_log_tokens(full_text, line_no, DART_LOG_TOKENS):
                      continue
                  violations.append(
                      f'[no_bare_catch] {file_path}:{line_no}: {content.strip()[:120]} '
                      f'(bare-catch without log/rethrow)'
                  )
                  break
      return violations


  def scan_python_added(
      added: List[Tuple[int, str]],
      full_text: str,
      file_path: str,
  ) -> List[str]:
      violations: List[str] = []
      lines = full_text.splitlines()
      for line_no, content in added:
          if _has_valid_override(content, is_python=True) or _override_in_preceding(lines, line_no, is_python=True):
              continue
          for pat in PY_BARE_EXCEPT:
              if pat.search(content):
                  if has_surrounding_log_tokens(full_text, line_no, PY_LOG_TOKENS):
                      continue
                  violations.append(
                      f'[no_bare_catch] {file_path}:{line_no}: {content.strip()[:120]} '
                      f'(bare-except without log/raise)'
                  )
                  break
      return violations


  def read_staged_file(file_path: str, repo_root: Optional[Path] = None) -> str:
      """Read the staged (index) content of a file (post-diff view)."""
      cmd = ['git', 'show', f':{file_path}']
      kwargs = {'capture_output': True, 'text': True, 'check': False}
      if repo_root is not None:
          kwargs['cwd'] = str(repo_root)
      r = subprocess.run(cmd, **kwargs)
      if r.returncode != 0:
          # Fallback to working tree (un-staged or unknown file)
          path = Path(repo_root or '.') / file_path
          if path.exists():
              return path.read_text(encoding='utf-8', errors='ignore')
          return ''
      return r.stdout


  def process_file(file_path: str, repo_root: Optional[Path] = None) -> List[str]:
      if is_exempt_path(file_path):
          return []
      added = get_added_lines(file_path, repo_root=repo_root)
      if not added:
          return []
      full_text = read_staged_file(file_path, repo_root=repo_root)
      if file_path.endswith('.dart'):
          return scan_dart_added(added, full_text, file_path)
      if file_path.endswith('.py'):
          return scan_python_added(added, full_text, file_path)
      return []


  def list_staged_files(repo_root: Optional[Path] = None) -> List[str]:
      cmd = ['git', 'diff', '--staged', '--name-only', '--diff-filter=AM',
             '--', '*.dart', '*.py']
      kwargs = {'capture_output': True, 'text': True, 'check': False}
      if repo_root is not None:
          kwargs['cwd'] = str(repo_root)
      r = subprocess.run(cmd, **kwargs)
      return [ln for ln in r.stdout.splitlines() if ln.strip()]


  def main() -> int:
      ap = argparse.ArgumentParser(
          description='GUARD-02 — diff-only bare-catch lint (Dart + Python).'
      )
      ap.add_argument('--staged', action='store_true',
                      help='Scan all staged Dart+Python files (default lefthook mode).')
      ap.add_argument('--file', nargs='*', default=[],
                      help='Scan specific files (for self-test + pytest).')
      ap.add_argument('--repo-root', default=None,
                      help='Git repo root for subprocess calls (test harness).')
      args = ap.parse_args()

      repo_root = Path(args.repo_root) if args.repo_root else None

      if args.staged:
          files = list_staged_files(repo_root=repo_root)
      elif args.file:
          files = args.file
      else:
          ap.error('must pass --staged or --file <paths>')
          return 2

      all_violations: List[str] = []
      for f in files:
          all_violations.extend(process_file(f, repo_root=repo_root))

      for v in all_violations:
          print(v, file=sys.stderr)

      if all_violations:
          print(
              f'[no_bare_catch] FAIL — {len(all_violations)} bare-catch violation(s) '
              f'introduced by this diff. Add logging/raise or `// lefthook-allow:bare-catch: <reason>`.',
              file=sys.stderr,
          )
          return 1
      return 0


  if __name__ == '__main__':
      sys.exit(main())
  ```

  """GUARD-02 pytest coverage — 12 cases covering D-05/D-06/D-07."""
  from __future__ import annotations
  import subprocess
  import sys
  from pathlib import Path

  ROOT = Path(__file__).resolve().parents[2]
  sys.path.insert(0, str(ROOT / "tools" / "checks"))

  import no_bare_catch as lint  # noqa: E402


  def _stage(repo: Path, rel: str, content: str) -> None:
      """Write and `git add` a file in the repo."""
      p = repo / rel
      p.parent.mkdir(parents=True, exist_ok=True)
      p.write_text(content, encoding="utf-8")
      subprocess.run(["git", "-C", str(repo), "add", rel], check=True)


  def _run_lint(repo: Path, files: list[str]) -> tuple[int, list[str]]:
      violations: list[str] = []
      for f in files:
          violations.extend(lint.process_file(f, repo_root=repo))
      return (1 if violations else 0, violations)


  # ─── D-05 bare-catch detection ────────────────────────────────────────────

      assert rc == 1
      assert any("bare-catch" in v for v in vios)
