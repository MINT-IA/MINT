  ```python
  #!/usr/bin/env python3
  """GUARD-06 — fallback proof-of-read via commit-msg hook.

  Per CONTEXT 34-CONTEXT.md D-16, D-17, D-18: Claude-coauthored commits must
  reference a `.planning/phases/<phase>/<padded>-READ.md` file (D-16 trailer
  convention) that lists the files the agent consulted (D-18 bullet format).

  Runs on the `commit-msg` lefthook hook — lefthook passes the path to
  `.git/COMMIT_EDITMSG` via `{1}` placeholder. This is the ONE commit-msg
  hook allowed in Phase 34 (D-04 amended — see Plan 34-05 SUMMARY for the
  explicit amendment rationale, per RESEARCH §Open Question 1 Option A).

  Human commits (no `Co-Authored-By: Claude` trailer) bypass automatically
  (D-17). MINT's convention is most commits DO carry the Claude trailer
  (RESEARCH A3), so most commits are effectively gated.

  Security: `Read:` path must begin with `.planning/phases/` to mitigate
  T-34-SPOOF-01 (attacker could point at `/dev/null` or unrelated files).

  Exit codes:
    0 — pass (human commit OR valid proof-of-read)
    1 — fail (Claude trailer present but Read: missing/invalid)

  Technical English only — dev-facing, no i18n (Pitfall 8).
  """
  from __future__ import annotations
  import argparse
  import re
  import sys
  from pathlib import Path

  TRAILER_CLAUDE = re.compile(r'^Co-Authored-By:\s+Claude', re.MULTILINE)
  TRAILER_READ = re.compile(r'^Read:\s+(\S+)\s*$', re.MULTILINE)
  ALLOWED_READ_PREFIX = '.planning/phases/'


  def check_commit_msg(msg: str, repo_root: Path) -> tuple[int, list[str]]:
      """Validate a commit message for GUARD-06 compliance.

      Returns (exit_code, messages).
      """
      messages: list[str] = []

      # Empty / whitespace-only → pass (nothing to check).
      if not msg.strip():
          return 0, []

      if not TRAILER_CLAUDE.search(msg):
          # D-17 human bypass.
          return 0, ['[proof_of_read] OK — human commit (no Claude trailer), bypass']

      match = TRAILER_READ.search(msg)
      if not match:
          messages.append(
              '[proof_of_read] FAIL — Claude-coauthored commit missing `Read:` trailer.'
          )
          messages.append(
              '  Required format: `Read: .planning/phases/<phase>/<padded>-READ.md`'
          )
          messages.append('  Per CONTEXT 34-CONTEXT.md D-16.')
          return 1, messages

      read_path_str = match.group(1).strip()

      # T-34-SPOOF-01 mitigation: path must point inside .planning/phases/
      if not read_path_str.startswith(ALLOWED_READ_PREFIX):
          messages.append(
              f'[proof_of_read] FAIL — Read: path must start with '
              f'`{ALLOWED_READ_PREFIX}`, got: {read_path_str}'
          )
          return 1, messages

      read_path = repo_root / read_path_str
      if not read_path.exists():
          messages.append(
              f'[proof_of_read] FAIL — Read: path does not exist on disk: {read_path_str}'
          )
          return 1, messages

      # D-18 format check: at least one bullet line starting with `- `
      content = read_path.read_text(encoding='utf-8', errors='ignore')
      has_bullet = any(
          line.strip().startswith('- ')
          for line in content.splitlines()
      )
      if not has_bullet:
          messages.append(
              f'[proof_of_read] FAIL — {read_path_str} has no `- <path>` bullet entries '
              f'(D-18 format: `- <path> — <why read>`).'
          )
          return 1, messages

      num_bullets = sum(
          1 for line in content.splitlines()
          if line.strip().startswith('- ')
      )
      messages.append(
          f'[proof_of_read] OK — Claude commit references {read_path_str} '
          f'({num_bullets} files listed)'
      )
      return 0, messages


  def main() -> int:
      ap = argparse.ArgumentParser(
          description='GUARD-06 proof-of-read commit-msg hook (D-16/D-17/D-18).'
      )
      ap.add_argument(
          '--commit-msg-file',
          required=True,
          help='.git/COMMIT_EDITMSG path (lefthook {1} placeholder)',
      )
      ap.add_argument(
          '--repo-root',
          default='.',
          help='Git repo root (default cwd; tests pass tmp dir)',
      )
      args = ap.parse_args()

      msg = Path(args.commit_msg_file).read_text(encoding='utf-8', errors='ignore')
      rc, messages = check_commit_msg(msg, Path(args.repo_root).resolve())
      for m in messages:
          stream = sys.stderr if '[proof_of_read] FAIL' in m else sys.stdout
          print(m, file=stream)
      return rc


  if __name__ == '__main__':
      sys.exit(main())
  ```
