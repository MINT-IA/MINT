  ```yaml
  # .github/workflows/bypass-audit.yml — Phase 34 GUARD-07 D-21/D-22.
  #
  # Secondary awareness tool for LEFTHOOK_BYPASS usage patterns.
  # PRIMARY ground-truth for bypass-induced regressions:
  # .github/workflows/lefthook-ci.yml (D-24, Plan 34-07) — that job re-runs
  # all lints on every PR and fails loud on regressions.
  #
  # This job greps commit bodies on `dev` for `LEFTHOOK_BYPASS` references
  # over the last 7 days and opens a `bypass-audit`-labelled issue if the
  # count exceeds 3 (D-22 threshold). Manual triage — no bot auto-fix.
  #
  # Scheduled Monday 09:00 UTC + post-merge to dev per D-21.
  name: Bypass audit (GUARD-07)

  on:
    schedule:
      # Monday 09:00 UTC per D-21
      - cron: '0 9 * * 1'
    push:
      branches: [dev]

  jobs:
    audit:
      name: Count bypass signals in last 7 days
      runs-on: ubuntu-latest
      permissions:
        contents: read
        issues: write
      steps:
        - uses: actions/checkout@v4
          with:
            fetch-depth: 0
        - name: Count LEFTHOOK_BYPASS references in commit bodies on dev (7d window)
          id: count
          run: |
            # D-21: grep commit bodies on dev for LEFTHOOK_BYPASS / [bypass:] markers
            # (voluntary signals per RESEARCH Open Question 6).
            git fetch origin dev --depth=0 2>/dev/null || true
            # Use --remotes in case we're not on dev locally
            count=$(git log --since="7 days ago" --pretty=%B origin/dev 2>/dev/null \
              | grep -c -E 'LEFTHOOK_BYPASS|\[bypass:' || true)
            echo "count=${count}" >> "$GITHUB_OUTPUT"
            echo "[bypass-audit] Found ${count} LEFTHOOK_BYPASS/[bypass:] references on dev (7d window)"
        - name: Create issue if threshold exceeded
          if: ${{ fromJSON(steps.count.outputs.count) > 3 }}
          uses: actions/github-script@v7
          with:
            script: |
              const count = '${{ steps.count.outputs.count }}';
              const title = `bypass-audit: ${count} LEFTHOOK_BYPASS references this week`;
              const body = [
                `Weekly audit (GUARD-07 D-21) detected **${count} commits** with`,
                `\`LEFTHOOK_BYPASS\` or \`[bypass:\` markers on \`dev\` in the last 7`,
                `days (threshold=3 per CONTEXT.md D-22).`,
                ``,
                `Manual triage:`,
                '',
                '```bash',
                'git log --since="7 days ago" --grep="LEFTHOOK_BYPASS\\|\\[bypass:" origin/dev',
                '```',
                ``,
                `**Note:** this audit is a secondary awareness tool. The primary`,
                `ground-truth bypass detector is \`.github/workflows/lefthook-ci.yml\``,
                `(GUARD-07 D-24) which re-runs all lints on every PR. If that job`,
                `passes on the PRs in question, the bypasses did not introduce`,
                `regressions — only the usage pattern is being flagged here.`,
              ].join('\n');
              await github.rest.issues.create({
                owner: context.repo.owner,
                repo: context.repo.repo,
                title: title,
                labels: ['bypass-audit'],
                body: body,
              });
              core.info(`[bypass-audit] issue created — count=${count}`);
  ```
