  ```python
  #!/usr/bin/env python3
  """GUARD-05 — ARB key + placeholder parity across 6 language files.

  Per CONTEXT 34-CONTEXT.md D-13, D-14: fail on any non-@ key-set divergence
  across {fr, en, de, es, it, pt} or ICU placeholder NAME drift in a value.
  FR template (per apps/mobile/l10n.yaml: template-arb-file: app_fr.arb) is
  the source of truth for placeholder metadata.

  Zero external deps — stdlib json + re only (D-14 "pas de dep ICU/intl").
  Per D-15: dead Dart orphan keys are out of scope — this lint checks parity
  CROSS-LANGUAGE only.

  Exit codes:
    0 — parity OK
    1 — divergence (stderr lists missing keys + placeholder drifts)

  Technical English — dev-facing diagnostics, no i18n (Pitfall 8).
  """
  from __future__ import annotations
  import argparse
  import json
  import re
  import sys
  from pathlib import Path
  from typing import Dict, List, Set, Tuple

  LANGS = ['fr', 'en', 'de', 'es', 'it', 'pt']

  # Default production L10N dir (overridable via --dir for tests)
  DEFAULT_L10N_DIR = Path(__file__).resolve().parents[2] / 'apps' / 'mobile' / 'lib' / 'l10n'

  # RESEARCH §Pattern 4 ICU placeholder name regex
  ICU_NAME = re.compile(
      r'\{\s*([A-Za-z_][A-Za-z0-9_]*)\s*(?:[},]|\s)',
      re.UNICODE,
  )
  ICU_KEYWORDS = {
      'plural', 'select', 'number', 'DateTime', 'date', 'time', 'ordinal',
  }


  def extract_placeholders(value: str) -> Set[str]:
      """Return placeholder name set from an ARB value string.

      Handles simple {name}, typed {name, number}, plural {n, plural, ...},
      select {sex, select, ...}. ICU keywords are filtered out.
      Per RESEARCH §Pitfall 3: plural inner `{# items}` are NOT matched as
      placeholders because `{` followed by `#` or digit is not matched by
      `[A-Za-z_]`, and inner `{count}` re-uses the outer `count` so the set
      dedupes correctly.
      """
      names: Set[str] = set()
      for m in ICU_NAME.finditer(value):
          token = m.group(1)
          if token in ICU_KEYWORDS:
              continue
          names.add(token)
      return names


  def load_arb(path: Path) -> Dict:
      """Read and parse an ARB file (JSON). Fail-fast on malformed input."""
      if not path.exists():
          raise FileNotFoundError(path)
      with path.open(encoding='utf-8') as f:
          return json.load(f)


  def check_parity(l10n_dir: Path) -> Tuple[int, List[str]]:
      """Verify key + placeholder parity across the 6 ARB files.

      Returns (exit_code, messages). exit_code 0 = pass, 1 = fail.
      """
      messages: List[str] = []
      data: Dict[str, Dict] = {}
      for lang in LANGS:
          p = l10n_dir / f'app_{lang}.arb'
          if not p.exists():
              messages.append(f'[arb_parity] FAIL — missing file: {p}')
              return 1, messages
          try:
              data[lang] = load_arb(p)
          except json.JSONDecodeError as e:
              messages.append(f'[arb_parity] FAIL — malformed JSON in {p}: {e}')
              return 1, messages

      # D-13 non-@ key parity
      keysets = {lang: {k for k in d if not k.startswith('@')} for lang, d in data.items()}
      union = set().union(*keysets.values())
      fail = False
      for lang, ks in keysets.items():
          missing = union - ks
          if missing:
              for k in sorted(missing)[:10]:
                  messages.append(
                      f"[arb_parity] FAIL — key '{k}' missing in app_{lang}.arb"
                  )
              if len(missing) > 10:
                  messages.append(
                      f"  ... and {len(missing) - 10} more in app_{lang}.arb"
                  )
              fail = True

      # D-13 placeholder parity (FR template is source-of-truth)
      fr = data['fr']
      for at_key, meta in fr.items():
          if not at_key.startswith('@') or not isinstance(meta, dict):
              continue
          phs = meta.get('placeholders')
          if not phs:
              continue
          plain_key = at_key[1:]
          expected = set(phs.keys())
          # Check each non-template language
          for lang in LANGS:
              value = data[lang].get(plain_key)
              if not value or not isinstance(value, str):
                  continue  # key-missing already flagged above
              actual = extract_placeholders(value)
              if actual != expected:
                  missing_ph = expected - actual
                  extra_ph = actual - expected
                  msg = (
                      f"[arb_parity] FAIL — key '{plain_key}' "
                      f"placeholder drift in app_{lang}.arb"
                  )
                  if missing_ph:
                      msg += f" missing={sorted(missing_ph)}"
                  if extra_ph:
                      msg += f" extra={sorted(extra_ph)}"
                  messages.append(msg)
                  fail = True

      if fail:
          return 1, messages
      messages.append(
          f"[arb_parity] OK — 6 ARB files parity OK "
          f"(non-@ keys={len(union)}, checked placeholders across "
          f"{sum(1 for _, m in fr.items() if isinstance(m, dict) and m.get('placeholders'))} @keys)"
      )
      return 0, messages


  def main() -> int:
      ap = argparse.ArgumentParser(
          description='GUARD-05 — ARB parity lint (6 langs, stdlib only).'
      )
      ap.add_argument(
          '--dir',
          default=str(DEFAULT_L10N_DIR),
          help='Directory containing app_{fr,en,de,es,it,pt}.arb (default: production l10n/)',
      )
      args = ap.parse_args()

      rc, messages = check_parity(Path(args.dir))
      for m in messages:
          stream = sys.stderr if '[arb_parity] FAIL' in m else sys.stdout
          print(m, file=stream)
      return rc


  if __name__ == '__main__':
      sys.exit(main())
  ```

  import arb_parity as lint  # noqa: E402

  def test_fixture_placeholder_drift_fail(fixtures_dir: Path):
      # Note: arb_drift_placeholder only has fr + en; lint expects all 6.
      # Create a minimal 6-file scenario inline for this test.
      # Skipped directory lookup approach — use inline tmp instead.
      import json, tempfile
      with tempfile.TemporaryDirectory() as tmp:
          tmp_dir = Path(tmp)
          fr_content = {
              "@@locale": "fr",
              "welcome": "Bienvenue {name}",
              "@welcome": {"placeholders": {"name": {"type": "String"}}},
          }
          en_content = {"@@locale": "en", "welcome": "Welcome {n}"}  # drift!
          other = lambda code: {"@@locale": code, "welcome": "ok {name}"}
          (tmp_dir / "app_fr.arb").write_text(json.dumps(fr_content), encoding="utf-8")

              (tmp_dir / f"app_{code}.arb").write_text(
                  json.dumps(other(code)), encoding="utf-8",
              )
          rc, msgs = lint.check_parity(tmp_dir)
          assert rc == 1
          assert any("welcome" in m and "app_en.arb" in m for m in msgs)


  def test_extract_placeholders_simple():
      assert lint.extract_placeholders("Bonjour {name}") == {"name"}


  def test_extract_placeholders_plural():
      # Per Pitfall 3: `item`/`items` are literals inside plural variants,
      # not placeholder names. Inner `{count}` is the same name as outer.
      result = lint.extract_placeholders(
          "{count, plural, one {1 item} other {{count} items}}"
      )
      assert result == {"count"}, f"expected {{'count'}}, got {result}"


  def test_extract_placeholders_select():
      result = lint.extract_placeholders(
          "{sex, select, male {il} female {elle} other {iel}}"
      )
      assert result == {"sex"}, f"expected {{'sex'}}, got {result}"


  def test_extract_placeholders_typed():
      result = lint.extract_placeholders("{amount, number, currency}")
      assert result == {"amount"}, f"expected {{'amount'}}, got {result}"


  def test_extract_placeholders_multiple():
      assert lint.extract_placeholders("{a} and {b}") == {"a", "b"}


  def test_extract_placeholders_empty():
      assert lint.extract_placeholders("no placeholders here") == set()


  def test_production_arb_files_parity():
