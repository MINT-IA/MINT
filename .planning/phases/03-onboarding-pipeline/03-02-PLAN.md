    ```dart
    Future<void> _onChipTap(BuildContext context, _IntentChip chip) async {
      final l10n = S.of(context)!;

      AnalyticsService().trackCTAClick(
        'intent_chip_tapped',
        screenName: '/onboarding/intent',
        data: {'chipKey': chip.chipKey, 'label': chip.label},
      );

      // 1. Persist onboarding state
      await ReportPersistenceService.setMiniOnboardingCompleted(true);
      await ReportPersistenceService.setSelectedOnboardingIntent(chip.chipKey); // chipKey, NOT label (per D-03 + Pitfall 2)

      // 2. Resolve intent mapping
      final mapping = IntentRouter.forChipKey(chip.chipKey);

      // 3. Compute premier eclairage (per D-03, D-08)
      if (mapping != null) {
        // Build a MinimalProfileResult — may be zero-valued if QuickStart not done yet
        // ChiffreChocSelector handles pedagogical fallback for zero-valued profiles (D-08)
        final profile = _buildMinimalProfile(context);
        final choc = ChiffreChocSelector.select(profile, stressType: mapping.stressType);

        if (choc != null) {
          // 4. Persist premier eclairage snapshot — display fields ONLY, no PII (T-03-02)
          await ReportPersistenceService.savePremierEclairageSnapshot({
            'value': choc.value,
            'title': choc.title,
            'subtitle': choc.subtitle,
            'colorKey': choc.colorKey,
            'suggestedRoute': mapping.suggestedRoute,
            'confidenceMode': choc.confidenceMode,
          });
        }

        // 5. Seed CapMemoryStore with goalIntentTag (per D-05)
        final memory = await CapMemoryStore.load();
        final updated = memory.copyWith(declaredGoals: [mapping.goalIntentTag]);
        await CapMemoryStore.save(updated);

        // 6. Seed CapSequenceEngine (per D-05)
        // Capture l10n before potential navigation
        CapSequenceEngine.build(
          profile: profile,
          memory: updated,
          goalIntentTag: mapping.goalIntentTag,
          l: l10n,
        );
      }

      if (!context.mounted) return;

      // 7. Build coach payload (preserved from current behavior)
      final payload = CoachEntryPayload(
        source: CoachEntrySource.onboardingIntent,
        userMessage: chip.message,
      );
      context.read<CoachEntryPayloadProvider>().setPayload(payload);

      // 8. Navigate to Aujourd'hui tab (per D-03: tab=0, NOT tab=1)
      context.go('/home?tab=0');
    }
    ```
