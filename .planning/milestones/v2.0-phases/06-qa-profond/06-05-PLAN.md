         a. Decode the test image and golden image as raw RGBA pixels
         b. Count pixels where any channel differs by more than a small per-pixel tolerance (e.g., 2 units out of 255)
         c. Compute `diffRatio = differentPixels / totalPixels`
         d. Return true (pass) if `diffRatio <= 0.015` (1.5%)

       - Use Flutter's built-in `matchesGoldenFile` image comparison infrastructure. The simplest approach:
         ```dart
         class TolerantGoldenFileComparator extends LocalFileComparator {
           TolerantGoldenFileComparator(super.testFile, {this.tolerance = 0.015});
           final double tolerance;

           @override
           Future<bool> compare(Uint8List imageBytes, Uri golden) async {
             final goldenFile = getGoldenFile(golden);
             if (!goldenFile.existsSync()) {
               // First run -- update goldens
               await update(golden, imageBytes);
               return true;
             }
             final goldenBytes = goldenFile.readAsBytesSync();
             final diffPixels = _countDifferentPixels(imageBytes, goldenBytes);
             final totalPixels = imageBytes.length ~/ 4; // RGBA
             final ratio = diffPixels / totalPixels;
             if (ratio > tolerance) {
               throw FlutterError(
                 'Golden "$golden" differs by ${(ratio * 100).toStringAsFixed(2)}% '
                 '(threshold: ${(tolerance * 100).toStringAsFixed(1)}%). '
                 '$diffPixels of $totalPixels pixels differ.',
               );
             }
             return true;
           }
         }
         ```
